NON-TERMINALS

start ::= statement_list <EOF>
expression ::= multiplicativeExpression ( plusexpression | minusexpression )*
plusexpression ::= <PLUS> multiplicativeExpression
minusexpression ::= <MINUS> multiplicativeExpression
multiplicativeExpression ::= unaryExpression ( timesterm | divterm )*
timesterm ::= ( <TIMES> unaryExpression )
divterm ::= ( <DIVIDE> unaryExpression )
unaryExpression ::= <MINUS> numberExpression
| numberExpression
numberExpression ::= varname
| number
| <LBRACKET> expression <RBRACKET>
varname ::= <VARNAME>
number ::= <NUMBER>
fndefine_statement ::= <PROC> ( fn_name <LBRACKET> ( var_list )? <RBRACKET> ) inside_fn_statement_list <END>
var_list ::= varname ( <COMMA> varname )*
statement_list ::= ( statement )*
inside_fn_statement_list ::= ( inside_fn_statement )*
inside_fn_statement ::= fncall_statement
| make_statement
| fd_statement
| rt_statement
| rpt_statement
statement ::= inside_fn_statement
| fndefine_statement
rpt_statement ::= <RPT> expression <LSQR> inside_fn_statement_list <RSQR>
fncall_statement ::= fn_name <LBRACKET> ( expression_list )? <RBRACKET>
expression_list ::= expression ( <COMMA> expression )*
fn_name ::= <ID>
fd_statement ::= <FD> expression
rt_statement ::= <RT> expression
make_statement ::= <MAKE> vardefn expression
vardefn ::= <VARNAME>
